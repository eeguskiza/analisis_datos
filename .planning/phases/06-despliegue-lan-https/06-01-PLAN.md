### Caddyfile.prod (versión final)

```caddyfile
# caddy/Caddyfile.prod — producción LAN
# Cambios clave vs caddy/Caddyfile dev:
#   1. SIN `auto_https disable_redirects` — queremos que Caddy inserte redirect 80->443.
#   2. Bloque con hostname explícito `nexo.ecsmobility.local` para que el cert
#      auto-emitido contenga el hostname estable que los usuarios bookmarkean.
#   3. Bloque `:443` se conserva como fallback cuando alguien escribe https://<IP_NEXO>.

nexo.ecsmobility.local {
    tls internal
    reverse_proxy web:8000

    log {
        output stdout
        format json
    }
}

# Fallback IP-directa (visitantes que escriben https://<IP_NEXO>).
# Cert autofirmado genérico; browser muestra warning pero el acceso funciona.
# Retirar cuando DNS interno esté operativo (deferred).
:443 {
    tls internal
    reverse_proxy web:8000
}
```

"""Validaciones estáticas de docker-compose.prod.yml (Phase 6 / Plan 06-01).

Dos niveles de test:
1. Parser YAML directo sobre docker-compose.prod.yml (sin Docker) —
   comprueba presencia de `!reset`, healthchecks, resource limits.
2. `docker compose config` merged (si Docker disponible) — comprueba
   que la fusión con el base produce ports vacíos y healthchecks
   correctos. Skip si docker no está instalado.
"""
from __future__ import annotations

import json
import shutil
import subprocess
from pathlib import Path

import pytest

REPO_ROOT = Path(__file__).resolve().parents[2]
OVERRIDE = REPO_ROOT / "docker-compose.prod.yml"


def _docker_available() -> bool:
    if shutil.which("docker") is None:
        return False
    try:
        r = subprocess.run(
            ["docker", "compose", "version"],
            capture_output=True, timeout=10, check=False,
        )
        return r.returncode == 0
    except Exception:
        return False


# ── Nivel 1: parse directo del archivo override (sin Docker) ───────────────

def test_override_exists():
    assert OVERRIDE.exists(), f"No existe {OVERRIDE}"


def test_override_mentions_reset_for_db_ports():
    content = OVERRIDE.read_text(encoding="utf-8")
    # Landmine 2: ports: [] NO resetea. Debe aparecer `!reset` junto a ports en db.
    assert "!reset" in content, (
        "Override debe usar `!reset` para borrar ports heredados (Landmine 2)."
    )
    # Verificación más estricta: el texto entre 'db:' y el siguiente service debe
    # contener 'ports: !reset'.
    db_block_start = content.find("  db:")
    assert db_block_start >= 0, "Falta service db: en el override."
    db_block = content[db_block_start:]
    next_service = db_block.find("\n  web:")
    if next_service > 0:
        db_block = db_block[:next_service]
    assert "ports: !reset" in db_block, (
        "`db.ports: !reset []` es obligatorio para cerrar 5432 al host (DEPLOY-02)."
    )


def test_override_web_healthcheck_curl_api_health():
    content = OVERRIDE.read_text(encoding="utf-8")
