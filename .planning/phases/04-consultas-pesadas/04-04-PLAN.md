      @router.post("/api/thresholds/{endpoint:path}/recalibrate")
      def recalibrate(endpoint: str, request: Request, db: DbNexo, confirm: bool = False):
          user = request.state.user
          # Últimos 30 runs ok|slow
          from nexo.data.models_nexo import NexoQueryLog
          rows = list(QueryLogRepo(db).iter_filtered(
              endpoint=endpoint,
              statuses=['ok','slow'],
              limit=30,
              order_desc=True,
          ))
          if len(rows) < 10:
              raise HTTPException(400, f"Solo hay {len(rows)} runs; se requieren al menos 10 para recalcular.")
          # Parse params_json para extraer n_recursos/n_dias (solo para pipeline/run).
          # Para bbdd/query/capacidad/operarios usar factor = median(actual_ms) puro.
          import json, statistics
          if endpoint == "pipeline/run":
              per_unit = []
              for r in rows:
                  try:
                      p = json.loads(r.params_json or "{}")
                      n_r = int(p.get("n_recursos", 0))
                      n_d = int(p.get("n_dias", 0))
                      if n_r > 0 and n_d > 0 and r.actual_ms > 500:  # Pitfall 6: filter outliers triviales
                          per_unit.append(r.actual_ms / (n_r * n_d))
                  except Exception:
                      continue
              if len(per_unit) < 10:
                  raise HTTPException(400, f"Solo {len(per_unit)} runs con params parseables + actual_ms > 500ms.")
              factor_nuevo = statistics.median(per_unit)
          else:
              durations = [r.actual_ms for r in rows if r.actual_ms > 0]
              if len(durations) < 10:
                  raise HTTPException(400, "Insuficientes runs válidos.")
              factor_nuevo = statistics.median(durations)
          if not confirm:
              return {"endpoint": endpoint, "factor_nuevo": factor_nuevo, "sample_size": len(rows), "preview": True}
          # Persistir
          current = ThresholdRepo(db).get(endpoint)
          if current is None:
              raise HTTPException(404, f"Threshold {endpoint} no existe")
          ThresholdRepo(db).update(
              endpoint=endpoint, warn_ms=current.warn_ms, block_ms=current.block_ms,
              factor_ms=factor_nuevo, updated_by=user.id, factor_touched=True,
          )
          db.commit()
          from nexo.services import thresholds_cache
          thresholds_cache.notify_changed(endpoint)
          return {"endpoint": endpoint, "factor_old": current.factor_ms, "factor_new": factor_nuevo, "sample_size": len(rows), "committed": True}
      ```

    def timeseries(self, *, endpoint: str, date_from: datetime, date_to: datetime) -> list[dict]:
        """Bucketiza por hour (<=7d) o day (>7d). Devuelve puntos {ts, estimated_ms, actual_ms}."""
        rango_days = (date_to - date_from).days
        granularity = "hour" if rango_days <= 7 else "day"
        stmt = text(f"""
            SELECT date_trunc('{granularity}', ts) AS bucket,
                   AVG(estimated_ms)::integer AS avg_est,
                   AVG(actual_ms)::integer AS avg_actual
            FROM nexo.query_log
            WHERE endpoint = :ep AND ts >= :df AND ts < :dt
            GROUP BY bucket
            ORDER BY bucket ASC
        """)
        rows = self._db.execute(stmt, {"ep": endpoint, "df": date_from, "dt": date_to}).all()
        return [{"ts": r.bucket.isoformat(), "estimated_ms": r.avg_est or 0, "actual_ms": r.avg_actual or 0} for r in rows]
    ```
