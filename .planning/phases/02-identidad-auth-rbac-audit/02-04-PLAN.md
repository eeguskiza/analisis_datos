## Decisiones técnicas heredadas del research

- **Research §Pattern 5**: `AuditMiddleware` lee `request.body()` (FastAPI 0.135.3 cachea — §Pitfall 2 no aplica), sanitiza con `_SENSITIVE_FIELDS = {"password", "pwd", "secret", "token", "clave", "contrasena"}`, respeta `_REDACTED_ENDPOINTS = {"/api/conexion/config", "/api/conexion/test"}` con `details=None`. Limita a 4096 chars.
- **Research §Pattern 5**: si `request.state.user is None` (request pública: `/login`, `/api/health`), el middleware **no escribe fila** — `return await call_next(request)` y listo. Queda coherente con "middleware registra cada request **autenticada**".
- **Research §Pattern 8 + §Pitfall 4**: el GRANT-only sobre `nexo.audit_log` se hizo en 02-01. Si el rol app es owner, el test IDENT-06 fallará (owner siempre puede DELETE). La tarea 4.4 decide entonces entre (a) crear rol dedicado `nexo_app` en un script extra o (b) aceptar el riesgo documentando en `docs/AUTH_MODEL.md`. **Default pragmático: probar primero**. Si el test pasa (muy probable en Postgres si `pg_user` no es owner del esquema), pasamos adelante. Si falla, ver Tarea 4.4 apartado "fallback".
- **Research §Pitfall 3**: `datetime.now(timezone.utc)` en `audit_log.ts`. Nunca `datetime.utcnow()`.
- **Research §Pitfall 1**: orden de middlewares. En 02-02 se registró sólo `AuthMiddleware`. Este plan añade la línea `app.add_middleware(AuditMiddleware)` **antes** de `app.add_middleware(AuthMiddleware)` para que Starlette ejecute auth primero (LIFO).
- **Research §Open Question 3**: paginación server-side en `/ajustes/auditoria` con `?page=N&limit=100` y `LIMIT/OFFSET`. Tabla Alpine en frontend sin paginación client-side (simple).
- **Research §Open Question 6**: `nexo.permissions` sembrado en 02-01 queda como catálogo read-only; `/ajustes/usuarios` NO lo consulta para permisos — la lógica está en código.

```python
# api/middleware/audit.py
import json
import logging
from datetime import datetime, timezone
from starlette.middleware.base import BaseHTTPMiddleware
from starlette.requests import Request

from nexo.db.engine import SessionLocalNexo
from nexo.db.models import NexoAuditLog

logger = logging.getLogger("nexo.audit")

_REDACTED_ENDPOINTS = {
    "/api/conexion/config",
    "/api/conexion/test",
}
_SENSITIVE_FIELDS = {"password", "pwd", "secret", "token", "clave", "contrasena"}
_MAX_DETAILS_CHARS = 4096


class AuditMiddleware(BaseHTTPMiddleware):
    async def dispatch(self, request: Request, call_next):
        user = getattr(request.state, "user", None)
        if user is None:
            # Request pública (/login, /api/health, /static/*): no se audita.
            return await call_next(request)

        details = None
        path = request.url.path
        if request.method in ("POST", "PUT", "PATCH"):
            if path in _REDACTED_ENDPOINTS:
                details = None
            else:
                try:
                    body_bytes = await request.body()
                    if body_bytes:
                        body_dict = json.loads(body_bytes)
                        if isinstance(body_dict, dict):
                            sanitized = {
                                k: ("[REDACTED]" if k.lower() in _SENSITIVE_FIELDS else v)
                                for k, v in body_dict.items()
                            }
                            details = json.dumps(sanitized, ensure_ascii=False)[:_MAX_DETAILS_CHARS]
                        else:
                            # payload no-dict (list, scalar): guardarlo acotado
                            details = json.dumps(body_dict, ensure_ascii=False)[:_MAX_DETAILS_CHARS]
                except Exception:
                    details = None   # body no JSON o parseo falló: no bloquea la request

        response = await call_next(request)

        try:
            with SessionLocalNexo() as db:
                db.add(NexoAuditLog(
                    ts=datetime.now(timezone.utc),
                    user_id=user.id,
                    ip=request.client.host if request.client else None,
                    method=request.method,
                    path=path,
                    status=response.status_code,
                    details_json=details,
                ))
                db.commit()
        except Exception:
            logger.exception("Error escribiendo nexo.audit_log")

        return response
```

```python
import csv
import io
from datetime import datetime
from fastapi import APIRouter, Depends, Query, Request
from fastapi.responses import HTMLResponse, StreamingResponse
from sqlalchemy import and_
from nexo.services.auth import require_permission
from nexo.db.engine import SessionLocalNexo
from nexo.db.models import NexoAuditLog, NexoUser
from api.deps import render

router = APIRouter(
    prefix="/ajustes/auditoria",
    tags=["ajustes"],
    dependencies=[Depends(require_permission("auditoria:read"))],   # solo propietario
)

def _apply_filters(q, user_email: str | None, date_from: str | None,
                   date_to: str | None, path: str | None, status: int | None):
    conditions = []
    if user_email:
        conditions.append(NexoUser.email == user_email)
    if date_from:
        conditions.append(NexoAuditLog.ts >= datetime.fromisoformat(date_from))
    if date_to:
        conditions.append(NexoAuditLog.ts <= datetime.fromisoformat(date_to))
    if path:
        conditions.append(NexoAuditLog.path.ilike(f"%{path}%"))
    if status:
        conditions.append(NexoAuditLog.status == status)
    if conditions:
        q = q.filter(and_(*conditions))
    return q

@router.get("", response_class=HTMLResponse)
async def listar(
    request: Request,
    user_email: str | None = Query(None),
    date_from: str | None = Query(None),
    date_to: str | None = Query(None),
    path: str | None = Query(None),
    status: int | None = Query(None),
    page: int = Query(1, ge=1),
    limit: int = Query(100, ge=1, le=500),
):
    with SessionLocalNexo() as db:
        q = db.query(NexoAuditLog, NexoUser).join(NexoUser, NexoUser.id == NexoAuditLog.user_id)
        q = _apply_filters(q, user_email, date_from, date_to, path, status)
        total = q.count()
        rows = q.order_by(NexoAuditLog.ts.desc()).offset((page - 1) * limit).limit(limit).all()
    return render("ajustes_auditoria.html", request, {
        "rows": rows, "total": total, "page": page, "limit": limit,
        "filters": {"user_email": user_email, "date_from": date_from,
                    "date_to": date_to, "path": path, "status": status},
    })

@router.get("/export")
async def export_csv(
    user_email: str | None = Query(None),
    date_from: str | None = Query(None),
    date_to: str | None = Query(None),
    path: str | None = Query(None),
    status: int | None = Query(None),
):
    def iter_csv():
        buf = io.StringIO()
        w = csv.writer(buf)
        w.writerow(["ts", "user_email", "ip", "method", "path", "status", "details_json"])
        yield buf.getvalue(); buf.seek(0); buf.truncate(0)
        with SessionLocalNexo() as db:
            q = db.query(NexoAuditLog, NexoUser).join(NexoUser, NexoUser.id == NexoAuditLog.user_id)
            q = _apply_filters(q, user_email, date_from, date_to, path, status)
            for row, user in q.yield_per(500):
                w.writerow([row.ts.isoformat(), user.email, row.ip,
                            row.method, row.path, row.status, row.details_json or ""])
                yield buf.getvalue(); buf.seek(0); buf.truncate(0)

    filename = f"nexo_audit_{datetime.utcnow().strftime('%Y%m%d_%H%M%S')}.csv"
    return StreamingResponse(
        iter_csv(),
        media_type="text/csv",
        headers={"Content-Disposition": f'attachment; filename="{filename}"'},
    )
```
